DROP DATABASE THESSOK;
CREATE DATABASE THESSOK;
USE THESSOK;
CREATE TABLE CATEGORY(
CLASSIFICATION CHAR(20) NOT NULL,
NAME CHAR(50) UNIQUE,
CATEGORY_NUM INT NOT NULL,
PRIMARY KEY(CATEGORY_NUM));

CREATE TABLE CUSTOMER(
CUSTOMER_NUM INT NOT NULL,
PASSWORD CHAR(15) NOT NULL,
ID CHAR(15) NOT NULL,
NAME CHAR(50) DEFAULT NULL,
AGE INT DEFAULT NULL, 
SEX CHAR DEFAULT NULL,
JOB CHAR(30) DEFAULT NULL,
ADDRESS CHAR(100) NOT NULL,
P_NUM CHAR(15) NOT NULL,
DESTINATION CHAR(100) DEFAULT NULL,
PRIMARY KEY(CUSTOMER_NUM));

CREATE TABLE SHOP( 
TEL_NO CHAR(15) NOT NULL,
LOCATION CHAR(100) NOT NULL,
NAME CHAR(20) NOT NULL,
SHOP_NUM INT NOT NULL,
PRIMARY KEY(SHOP_NUM));

CREATE TABLE ORDERS(
DATE CHAR(15) DEFAULT NULL,
PURCHASE_NUM INT NOT NULL,
CUS_NUM INT NOT NULL,
PRIMARY KEY(PURCHASE_NUM),
FOREIGN KEY(CUS_NUM) REFERENCES CUSTOMER(CUSTOMER_NUM));

CREATE TABLE PRODUCT(
PRODUCT_NUM INT NOT NULL,
NAME CHAR(30) NOT NULL,
PRICE INT NOT NULL,
IMPORTER CHAR(50) NOT NULL,
SHOP_NUM INT NOT NULL,
EXPIRATION_DATE CHAR(15) NOT NULL,
DATE_OF_MAN CHAR(15) NOT NULL,
SHOP_STOCK INT NOT NULL,
STOCK INT NOT NULL,
U_CATEGORY_NUM INT NOT NULL,
D_CATEGORY_NUM INT NOT NULL,
PRIMARY KEY(PRODUCT_NUM,shop_num),
FOREIGN KEY(U_CATEGORY_NUM) REFERENCES CATEGORY(CATEGORY_NUM),
FOREIGN KEY(D_CATEGORY_NUM) REFERENCES CATEGORY(CATEGORY_NUM));

CREATE TABLE PRODUCTION(
PRODUCT_NUM INT NOT NULL,
PRODUCER CHAR(30) NOT NULL,
LOCATION CHAR(100) NOT NULL,
PRIMARY KEY (PRODUCT_NUM),
FOREIGN KEY(PRODUCT_NUM) REFERENCES PRODUCT(PRODUCT_NUM));

CREATE TABLE SHOPPING_CART(
CUSTOMER_NUM INT NOT NULL,
S_PRODUCT_NUM INT NOT NULL,
R_PURCHASE_NUM INT NOT NULL,
S_SHOP_NUM INT NOT NULL,
PRIMARY KEY(CUSTOMER_NUM, S_PRODUCT_NUM, R_PURCHASE_NUM, S_SHOP_NUM),
FOREIGN KEY(CUSTOMER_NUM) REFERENCES CUSTOMER(CUSTOMER_NUM),
FOREIGN KEY(S_PRODUCT_NUM) REFERENCES PRODUCT(PRODUCT_NUM),
FOREIGN KEY(R_PURCHASE_NUM) REFERENCES ORDERS(PURCHASE_NUM),
FOREIGN KEY(S_SHOP_NUM) REFERENCES SHOP(SHOP_NUM));
